\begin{frame}{Solution}

Generates 4 loop invariants

\begin{itemize}
\tightlist
\item
  Starting bound
\item
  Equal array length
\item
  Upper bound
\item
  Array iterative assignment
\end{itemize}

\end{frame}

\begin{frame}[fragile]{Starting Bound}

\begin{itemize}
\tightlist
\item
  Generally counting variables
\end{itemize}

\begin{verbatim}
int i = 0
while C
    where i >= 0
    i = i + 1
\end{verbatim}

\end{frame}

\begin{frame}[fragile]{Equal array length}

\begin{verbatim}
copy = items
// or
copy = [0; |items|]

while i <= |items|
    where |copy| == |items|:
    copy[i] = f(items[i])
    i = i + 1
\end{verbatim}

\end{frame}

\begin{frame}[fragile]{Upper bound}

\begin{verbatim}
int i = 0
while i < C
    where i <= C:
    i = i + 1
\end{verbatim}

\end{frame}

\begin{frame}[fragile]{Array iterative assignment}

\begin{verbatim}
int i = 0
while i <= |items|
    where all { k in 0..i | items[k] == f(k) }:
    items[i] = f(i)
    i = i + 1
\end{verbatim}

\end{frame}
